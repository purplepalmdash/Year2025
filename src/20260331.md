### 1. ubuntu22.04/24.04 aosp building

```
sudo apt install -y \
    build-essential ccache libncurses6 libssl-dev m4 unzip zip brotli bc \
    bison curl flex g++-multilib gcc-multilib git gnupg gperf imagemagick \
    lib32ncurses-dev lib32readline-dev lib32z1-dev liblz4-tool libncurses-dev \
    libsdl1.2-dev libxml2 libxml2-utils lzop pngcrush rsync schedtool \
    squashfs-tools xsltproc zip zlib1g-dev adb fastboot python-is-python3 \
    abootimg git-lfs libtinfo6
# 方法一：添加 focal 仓库安装（推荐）
echo "deb http://security.ubuntu.com/ubuntu focal-security main universe" | sudo tee /etc/apt/sources.list.d/ubuntu-focal-sources.list
sudo dpkg --add-architecture i386
sudo apt update
sudo apt install libncurses5:i386 libncurses5
```
