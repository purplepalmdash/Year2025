### 1. rutabaga+gfxstream(lineageos)
Script:     

```
[dash@archgfx ~]$ cat run_rutabaga_vulkangfx_lineageos.sh 
#!/bin/bash
# QEMU + Rutabaga + Gfxstream 启动脚本 - LineageOS 版本

set -e

VM_NAME="venus_lineageos232"
DISK_PRIMARY="/var/lib/libvirt/images/houdini_lin232.qcow2"
DISK_SECONDARY="/var/lib/libvirt/images/houdini_lin232_1.qcow2"
OVMF_CODE="/usr/share/edk2/x64/OVMF_CODE.4m.fd"
OVMF_VARS="/var/lib/libvirt/qemu/nvram/testb608.fd"
MEMORY="8G"
CPUS="4"
HOSTMEM="4G"

# 显示设置
DISPLAY_TYPE="gtk,gl=on,show-cursor=on"

echo "🚀 启动 $VM_NAME (Rutabaga + Gfxstream + LineageOS)"

qemu-system-x86_64 \
    -name "$VM_NAME" \
    -machine type=q35,accel=kvm \
    -cpu host \
    -smp "$CPUS" \
    -m "$MEMORY" \
    -enable-kvm \
    -vga none \
    -display "$DISPLAY_TYPE" \
    -device virtio-gpu-rutabaga-pci,\
hostmem="$HOSTMEM",\
gfxstream-vulkan=on,\
cross-domain=on,\
wsi=surfaceless \
    \
    -drive if=pflash,format=raw,file="$OVMF_CODE",readonly=on \
    -drive if=pflash,format=raw,file="$OVMF_VARS" \
    \
    -drive file="$DISK_PRIMARY",format=qcow2,if=virtio,cache=writeback \
    -drive file="$DISK_SECONDARY",format=qcow2,if=virtio,cache=writeback \
    \
    -netdev user,id=net0,hostfwd=tcp::25555-:5555 \
    -device virtio-net-pci,netdev=net0 \
    \
    -device virtio-keyboard-pci \
    -device virtio-mouse-pci \
    -device usb-ehci,id=usb \
    -device usb-tablet,bus=usb.0 \
    \
    -audiodev pa,id=sound0 \
    -device intel-hda \
    -device hda-duplex,audiodev=sound0 \
    \
    -monitor stdio \
    -serial none
```
