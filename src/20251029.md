### 1. ipp-usb building(uos/arm64)
run docker instance for getting the source code:     

```
sudo docker run -it -v ~/ipp:/mnt8 debian:bullseye /bin/bash
In docker
apt update -y
apt install dpkg-dev
mkdir debhelper && cd debhelper 
 apt source debhelper
cd .. && mkdir ronn && cd ronn
apt source ronn
cd .. && mkdir goipp && cd goipp
apt source golang-github-openprinting-goipp-dev
cd ../ && mkdir ipp && cd ipp
apt source ipp-usb
cd ../ && mkdir gem2deb && cd gem2deb
apt source gem2deb
cd ..
tar czvf combine.tar.gz debhelper  goipp  ipp  ronn gem2deb
```
In uos vm , compile the deb:     

```
mkdir ipp
cd ipp
tar xzvf /home/test/combine.tar.gz
apt update
apt install build-essential devscripts fakeroot
cd gem2deb
 vim gem2deb-1.4/debian/rules 
cd gem2deb-1.4/
apt build-dep .
debuild -us -uc
cd ..
apt install ./gem2deb_1.4_arm64.deb ./gem2deb-test-runner_1.4_arm64.deb 

 cd ../ronn/ruby-ronn-0.9.1/
 apt build-dep .
 debuild -us -uc
 cd ..
 apt install ./ronn_0.9.1-2_all.deb ./ruby-ronn_0.9.1-2_all.deb 
 cd ../debhelper/debhelper-13.3.4/
 apt build-dep .
 cd ../../goipp/golang-github-openprinting-goipp-1.0.0/
 apt build-dep .
 debuild -us -uc
 cd ..
 apt install ./golang-github-openprinting-goipp-dev_1.0.0-1_all.deb 
 cd ../ipp/ipp-usb-0.9.17/
 apt build-dep .
 debuild -us -uc
 cd ..
 apt install ./ipp-usb_0.9.17-3_arm64.deb 

 cp ./gem2deb/gem2deb_1.4_arm64.deb ./gem2deb/gem2deb-test-runner_1.4_arm64.deb ./ipp/ipp-usb_0.9.17-3_arm64.deb ./debhelper/debhelper-13.3.4/dh_installdeb ./debhelper/debhelper-13.3.4/dh_builddeb ./goipp/golang-github-openprinting-goipp-dev_1.0.0-1_all.deb ~/uos_ipp_deb_arm64/
 scp -r uos_ipp_deb_arm64/ dash@192.168.1.213:~
```
