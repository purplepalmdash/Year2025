  -smp 24,sockets=1,cores=24,threads=1 \
  -m 64G \
  -drive file=./testflex1.qcow2,format=qcow2,if=none,id=virtio-disk0,cache=none,aio=native \
  -device virtio-blk-pci,drive=virtio-disk0,id=disk0,bootindex=1 \

  -netdev bridge,id=net0,br=virbr0 \
  -device virtio-net-pci,netdev=net0,mac=52:54:00:12:34:56 \
  -vga none \
  -nographic \
  -chardev stdio,id=char0,mux=on \
  -serial chardev:char0 \
  -mon chardev=char0 \
  -boot c

In host level, enable following.    

```
# 1. 在 FORWARD 链最前面插入规则（优先级最高）
sudo iptables -I FORWARD 1 -i virbr0 -j ACCEPT
sudo iptables -I FORWARD 1 -o virbr0 -m state --state RELATED,ESTABLISHED -j ACCEPT

# 2. 确保 NAT 规则也被优先处理
sudo iptables -t nat -I POSTROUTING 1 -s 192.168.122.0/24 ! -d 192.168.122.0/24 -j MASQUERADE
```
