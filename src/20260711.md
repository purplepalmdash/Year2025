### 2. run venus(qemu)
Command:      

```
$ cat run_ubuntu_venes.sh 
#!/bin/bash
# QEMU + virtio-gpu-gl + DRM Native Context 版本（推荐）

set -e

VM_NAME="ubuntu2604-drm"
DISK="/home/dash/ubuntu2604.qcow2"
MEMORY="8G"
CPUS="8"
HOSTMEM="4G"

echo "🚀 启动 $VM_NAME (virtio-gpu-gl + DRM Native Context)"

qemu-system-x86_64 \
    -name "$VM_NAME" \
    -machine type=q35,accel=kvm \
    -cpu host \
    -smp "$CPUS" \
    -m "$MEMORY" \
    -enable-kvm \
    -vga none \
    -display sdl,gl=on,show-cursor=on \
    -device virtio-gpu-gl,hostmem="$HOSTMEM",blob=on,venus=on \
    \
    -drive file="$DISK",format=qcow2,if=virtio,cache=writeback \
    \
    -netdev user,id=net0,hostfwd=tcp::2222-:22 \
    -device virtio-net-pci,netdev=net0 \
    \
    -device virtio-keyboard-pci \
    -device usb-ehci,id=usb \
    -device usb-tablet,bus=usb.0 \
    \
    -audiodev pa,id=sound0 \
    -device intel-hda \
    -device hda-duplex,audiodev=sound0 \
    \
    -monitor stdio \
    -serial none \
    "$@"
```
