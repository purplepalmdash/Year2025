### 2. docker-ce binary(euler)
Download the binary:      

```
wget https://download.docker.com/linux/static/stable/aarch64/docker-29.0.0.tgz
tar xvf .....
[root@euler docker-install]# cd docker/
[root@euler docker]# ls
containerd  containerd-shim-runc-v2  ctr  docker  dockerd  docker-init  docker-proxy  runc
[root@euler docker]# cp * /usr/bin/
[root@euler docker]# sudo chmod +x /usr/bin/docker* /usr/bin/containerd* /usr/bin/runc
[root@euler docker]# sudo groupadd docker || true
[root@euler docker]# sudo usermod -aG docker $USER
[root@euler docker]# newgrp docker
```
Create systemd items:    

```
[root@euler system]# cat docker.socket 
[Unit]
Description=Docker Socket for the API

[Socket]
ListenStream=/run/docker.sock
SocketMode=0660
SocketUser=root
SocketGroup=docker

[Install]
WantedBy=sockets.target
[root@euler system]# cat docker.service 
[Unit]
Description=Docker Application Container Engine
Documentation=https://docs.docker.com
After=network-online.target containerd.service
Wants=network-online.target
Requires=containerd.service docker.socket
StartLimitBurst=3
StartLimitIntervalSec=60

[Service]
Type=notify
ExecStart=/usr/bin/dockerd -H fd:// --containerd=/run/containerd/containerd.sock
ExecReload=/bin/kill -s HUP $MAINPID
TimeoutStartSec=0
RestartSec=2
Restart=always

LimitNPROC=infinity
LimitCORE=infinity
TasksMax=infinity
Delegate=yes
KillMode=process
OOMScoreAdjust=-500

[Install]
WantedBy=multi-user.target
[root@euler system]# cat containerd.service 
[Unit]
Description=containerd container runtime
Documentation=https://containerd.io
After=network.target local-fs.target

[Service]
ExecStartPre=-/sbin/modprobe overlay
ExecStart=/usr/bin/containerd

Type=notify
Delegate=yes
KillMode=process
Restart=always
RestartSec=5
LimitNPROC=infinity
LimitCORE=infinity
TasksMax=infinity
OOMScoreAdjust=-999

[Install]
WantedBy=multi-user.target

```
Start service:       

```
[root@euler system]# sudo systemctl enable --now docker.socket
Created symlink /etc/systemd/system/sockets.target.wants/docker.socket → /etc/systemd/system/docker.socket.
[root@euler system]# sudo systemctl enable --now docker.service
Created symlink /etc/systemd/system/multi-user.target.wants/docker.service → /etc/systemd/system/docker.service.
[root@euler system]# sudo systemctl enable --now containerd
Created symlink /etc/systemd/system/multi-user.target.wants/containerd.service → /etc/systemd/system/containerd.service.
[root@euler system]# docker images
IMAGE   ID             DISK USAGE   CONTENT SIZE   EXTRA
```
