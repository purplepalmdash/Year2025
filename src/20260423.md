### 2. build waydroid(aosp14)
Step:    

```
sudo apt remove --purge rustc cargo rustc-1.82 cargo-1.82 -y
sudo apt autoremove -y
rustup default stable
rustup update stable
rustc --version
cargo --version
cargo install bindgen-cli
bindgen --version
which bindgen
# default is the aosp's prebuilt
echo 'export PATH="$HOME/.cargo/bin:$PATH"' >> ~/.bashrc
source ~/.bashrc 
which bindgen
bindgen --version
# this time we use the upgraded bindgen
rustup target add x86_64-linux-android
rustup target add i686-linux-android
make vendorimage -j12
```
